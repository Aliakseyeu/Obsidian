```php
class Cabinet extends ActiveRecord  
{  
    public function getAccounts(): ActiveQuery  
    {  
        return $this->hasMany(Account::class, ['id' => 'account_id'])  
            ->viaTable('account__cabinet', ['cabinet_id' => 'id']);  
    }  
}
```
