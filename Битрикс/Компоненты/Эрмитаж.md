Используется для управления контентом со страницы в режиме редактирования
```php
<?php  
    $this->AddEditAction(
	    $val['ID'],
	    $val['EDIT_LINK'], 
	    CIBlock::GetArrayByID($arParams["IBLOCK_ID"], "ELEMENT_EDIT")
	);  
    $this->AddDeleteAction(
	    $val['ID'],
	    $val['DELETE_LINK'], 
		CIBlock::GetArrayByID($arParams["IBLOCK_ID"], "ELEMENT_DELETE"), 
		array(
			"CONFIRM" => GetMessage(
				'FAQ_DELETE_CONFIRM', 
				array(
					"#ELEMENT#" => CIBlock::GetArrayByID(
						$arParams["IBLOCK_ID"], 
						"ELEMENT_NAME"
					)
				)
			)
		)
	);  
?>  

<div class="accordion-item"  id="<?=$this->GetEditAreaId($val['ID']);?>">  
</div>
```

```php
$iblockId = 7;  
  
$filter = [  
    '=IBLOCK_ID' => 7  
];  
  
$select = [  
    'ID',  
    'NAME'  
];  
  
$this->arResult['TAGS'] = \Bitrix\Iblock\ElementTable::getList([  
    'select' => $select,  
    'filter' => $filter  
])->fetchAll();  
  
foreach ($this->arResult['TAGS'] as &$tag) {  
    // Получаем ссылки из панели управления  
    $buttons = CIBlock::GetPanelButtons(  
        $iblockId,  
        $tag['ID'],  
        0,  
        ["SECTION_BUTTONS" => false, "SESSID" => false]  
    );  
  
    $tag['EDIT_LINK'] = $buttons["edit"]["edit_element"]["ACTION_URL"];  
    $tag['DELETE_LINK'] = $buttons["edit"]["delete_element"]["ACTION_URL"];  
  
    // Для визуального режима  
    $this->AddEditAction(  
        $tag['ID'],  
        $tag['EDIT_LINK'],  
        CIBlock::GetArrayByID($iblockId, "ELEMENT_EDIT")  
    );  
    $this->AddDeleteAction(  
        $tag['ID'],  
        $tag['DELETE_LINK'],  
        CIBlock::GetArrayByID($iblockId, "ELEMENT_DELETE"),  
        ["CONFIRM" => "Удалить элемент?"]  
    );  
}  
unset($tag);
```
