if($arResult['ID']){  
    $order = ['DATE_ACTIVE_FROM' => 'ASC', 'ID' =>  'ASC'];  
    // Аналог WHERE
    $filter = [  
        'IBLOCK_ID' =>  $arResult['IBLOCK_ID'],  
        'ACTIVE' => 'Y',  
        'CHECK_PERMISSIONS' => 'Y',  
    ];  
    if($arParams['CHECK_DATES']){  
        $filter['ACTIVE_DATE'] = 'Y';  
    }  
    $select = ['ID', 'IBLOCK_ID', 'NAME', 'DETAIL_PAGE_URL'];  
    // Параметры постраничной навигации (pagination). Аналог LIMIT в SQL
    $navParams = [  
        'nElementID' => $arResult['ID'],  // Текущий элемент
        'nPageSize' => 1,  // Количество на странице
    ];  
    // GetList Возвращает курсор
    $result = CIBlockElement::GetList(  
        $order,  
        $filter,  
        false,        
        $navParams,  
        $select  
    );  
  
    $foundCurrent = false;  
    while($item = $result->GetNext()){  
        if($item['ID'] == $arResult['ID']){  
            $foundCurrent = true;  
            continue;        }  
        $item['SHORT_NAME'] = TruncateText($item['NAME'], 50);  
        if(!$foundCurrent){  
            $arResult['PREVIOUS'] = $item;  
        } else {  
            $arResult['NEXT'] = $item;  
        }  
    }}
