protected static $oldGroups = [];  
  
public static function fillOldGroups(&$arParams)  
{  
    $groups = UserGroupTable::getList([  
        'filter' => [  
            'USER_ID' => $arParams['ID']  
        ],  
        'select' => ['GROUP_ID']  
    ]);  
  
    static::$oldGroups = [];  
  
    while ($group = $groups->fetch())  
    {        
	    static::$oldGroups[] = $group['GROUP_ID'];  
    }  
}
